119float GetShadowFactor( ShadowData shadowData )
120{
121 if( any(saturate(abs(shadowData.Position.xy)-shadowData.Position.w)) )
122 {
123 return 1.f;
124 }
125 else
126 {
127
128 float4 shadow = shadowData.Position / shadowData.Position.w;
129 float2 shadowTexCoords = shadow.xy * 0.5f + 0.5f;
130 shadowTexCoords.y = 1-shadowTexCoords.y;
131
132 float2 shadowMap = tex2D( ShadowMapSampler, shadowTexCoords ).rg;
133 float shadowMapDepth = shadowMap.r;
134 float shadowMapDepthSquared = shadowMap.g;
135
136 if(shadow.z > shadowMapDepth)
137 {
138 float var = shadowMapDepthSquared - shadowMapDepth*shadowMapDepth;
139 var = saturate( var + 0.0001 );
140
141 float d = shadow.z - shadowMapDepth;
142 float p = var / (var + d * d);
143
144 //p = pow(p, 5.0f);
145
146 return p;
147 }
148 else
149 {
150 return 1.f;
151 }
152 }
153}

201ShadowSplitInfo GetSplitInfo( ShadowData shadowData )
202{
203
204 float2 shadowTexCoords[NumSplits] =
205 {
206 shadowData.TexCoords_0_1.xy,
207 shadowData.TexCoords_0_1.zw,
208 shadowData.TexCoords_2_3.xy,
209 shadowData.TexCoords_2_3.zw
210 };
211
212 float lightSpaceDepth[NumSplits] =
213 {
214 shadowData.LightSpaceDepth.x,
215 shadowData.LightSpaceDepth.y,
216 shadowData.LightSpaceDepth.z,
217 shadowData.LightSpaceDepth.w,
218 };
219
220 for( int splitIndex=0; splitIndex < NumSplits; splitIndex++ )
221 {
222 if( shadowTexCoords[splitIndex].x >= TileBounds[splitIndex].x && shadowTexCoords[splitIndex].x <= TileBounds[splitIndex].y &&
223 shadowTexCoords[splitIndex].y >= TileBounds[splitIndex].z && shadowTexCoords[splitIndex].y <= TileBounds[splitIndex].w )
224 {
225 ShadowSplitInfo result;
226 result.TexCoords = shadowTexCoords[splitIndex];
227 result.LightSpaceDepth = lightSpaceDepth[splitIndex];
228 result.SplitIndex = splitIndex;
229
230 return result;
231 }
232 }
233
234 ShadowSplitInfo result = { float2(0,0), 0, NumSplits };
235 return result;
236}

252CascadeBlendingInfo GetBlendingInfo(ShadowData shadowData)
253{
254 float2 shadowTexCoords[NumSplits+1] =
255 {
256 shadowData.TexCoords_0_1.xy,
257 shadowData.TexCoords_0_1.zw,
258 shadowData.TexCoords_2_3.xy,
259 shadowData.TexCoords_2_3.zw,
260 float2(0,0)
261 };
262
263 CascadeBlendingInfo result = (CascadeBlendingInfo)0;
264 const float blendingBandSize = 16.0f / ShadowMapSize;
265
266 for( int i=0; i<NumSplits; ++i )
267 {
268 if( shadowTexCoords[i].x >= TileBounds[i].x &&
269 shadowTexCoords[i].x <= TileBounds[i].y &&
270 shadowTexCoords[i].y >= TileBounds[i].z &&
271 shadowTexCoords[i].y <= TileBounds[i].w )
272 {
273 float minDist = min(min(min(
274 shadowTexCoords[i].x - TileBounds[i].x,
275 TileBounds[i].y - shadowTexCoords[i].x),
276 shadowTexCoords[i].y - TileBounds[i].z),
277 TileBounds[i].w - shadowTexCoords[i].y);
278
279 result.BlendFactor.x = saturate(minDist / blendingBandSize);
280 result.BlendFactor.y = 1.0f - result.BlendFactor.x;
281
282 result.TexCoords.xy = shadowTexCoords[i];
283 result.TexCoords.zw = shadowTexCoords[i+1];
284
285 result.CascadeIndices.x = i;
286 result.CascadeIndices.y = i+1;
287
288 return result;
289 }
290 }
291
292 return result;
293}
